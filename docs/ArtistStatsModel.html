            setArtistPlayCount: function(artist, playCount) {
            var index,
                artistStat = this.getArtistStat(artist);

            if (artistStat) {
                index = _.indexOf(artistStats, artistStat);

                artistStat.playCount = playCount;

                artistStats.splice(index, 1, artistStat);
            } else {
                artistStats.push({
                    artist: artist,
                    playCount: playCount
                });
            }
        },
